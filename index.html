# TensorRT-Optimized Gaze & Skeleton Pose Pipeline
import tensorrt as trt
import numpy as np

class ProctorInferenceEngine:
    def __init__(self, engine_path):
        self.logger = trt.Logger(trt.Logger.WARNING)
        self.runtime = trt.Runtime(self.logger)
        with open(engine_path, "rb") as f:
            self.engine = self.runtime.deserialize_cuda_engine(f.read())
        self.context = self.engine.create_execution_context()

    def analyze_gaze_vectors(self, facial_landmarks, skeletal_joints):
        left_eye, right_eye = facial_landmarks[36:42], facial_landmarks[42:48]
        nose_bridge = facial_landmarks[27:31]
        head_pose = np.cross(left_eye[0] - right_eye[3], nose_bridge[0] - nose_bridge[-1])
        gaze_angle = np.dot(head_pose, skeletal_joints["shoulder_center"]) / (
            np.linalg.norm(head_pose) * np.linalg.norm(skeletal_joints["shoulder_center"])
        )
        return float(gaze_angle > 0.82) # Anomaly boundary flag

# Multi-Document RAG with Qdrant Hybrid Search & Re-ranking
from qdrant_client import QdrantClient
from sentence_transformers import CrossEncoder

class AdvancedRAGPipeline:
    def __init__(self):
        self.db = QdrantClient("http://localhost:6333")
        self.reranker = CrossEncoder("cross-encoder/ms-marco-MiniLM-L-6-v2")

    def query_vectors_with_rerank(self, query_str, top_k=15):
        candidates = self.db.search(
            collection_name="notes_collection",
            query_vector=self.encode_query(query_str),
            limit=top_k
        )
        pairs = [[query_str, c.payload["text_content"]] for c in candidates]
        scores = self.reranker.predict(pairs)
        ranked = sorted(zip(candidates, scores), key=lambda x: x[1], reverse=True)
        return [doc[0].payload for doc in ranked[:3]] # Return top 3 candidates
